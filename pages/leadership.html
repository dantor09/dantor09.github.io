    void setup() {
        pinMode(trigPin, OUTPUT);
        pinMode(echoPin, INPUT);
        pinMode(led1, OUTPUT);
        pinMode(led2, OUTPUT);
        pinMode(led3, OUTPUT);
        pinMode(led4, OUTPUT);
        pinMode(F_L, OUTPUT);
        pinMode(F_R, OUTPUT);
        pinMode(B_L, OUTPUT);
        pinMode(B_R, OUTPUT);
        Serial.begin(9600);
    }
    
    void loop() {
        impulse();
        distance = calc_distance();
        print_distance();
        digitalWrite(B_R, LOW);
        digitalWrite(B_R, LOW);
        delay(1000);
    
        analogWrite(B_R, 0);
        delay(5000);
    
        while (distance >= 25) {
            digitalWrite(led1, LOW);
            digitalWrite(led2, LOW);
            digitalWrite(led3, LOW);
            digitalWrite(led4, LOW);
    
            Serial.print("Starting ...");
            move_forward();
            impulse();
            distance = calc_distance();
            print_distance();
        }
        back_up();
        spin_counter();
    }
    
    void back_up()
    {
        digitalWrite(led1, HIGH);
        digitalWrite(led2,HIGH);
        digitalWrite(led3,HIGH);
        digitalWrite(led4,HIGH);
        digitalWrite(F_R,LOW);
        digitalWrite(F_L,LOW);
        digitalWrite(B_R,LOW);
        digitalWrite(B_R,LOW);
        digitalWrite(B_R,HIGH);
        digitalWrite(B_L,HIGH);
        
        delay(500);
    }
    void move_forward()
    {
        digitalWrite(B_L,LOW);
        digitalWrite(F_L,HIGH);
        digitalWrite(F_R,HIGH);
      
    }
    
    void spin_counter()
    {
        digitalWrite(F_R,LOW);
        digitalWrite(F_L,LOW);
        digitalWrite(B_R,LOW);
        digitalWrite(B_R,LOW);
        digitalWrite(F_R,HIGH);
        digitalWrite(B_L,HIGH);
        delay(1000);
    }
    void print_distance()
    {
      Serial.print("Distance ="); //print prints on monitor section as one line
      Serial.println(distance);  //println prints on monitor section as one line\n
      
    }
    
    float calc_distance()
    {
       duration = pulseIn(echoPin, HIGH); //Time it took wave to bounce back to Ultrasonic sensor
      distance = (duration/2) * 0.0343; //distance = Time * Velocity
      return distance;
    }
    void impulse()
    {
        digitalWrite(trigPin, LOW);
        delayMicroseconds(2);
        digitalWrite(trigPin, HIGH);
        delayMicroseconds(10);
        digitalWrite(trigPin, LOW);
      
    }
